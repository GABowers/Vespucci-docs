32 
33  //this method of determining type may not be valid on non-Win platforms

35 
36  bool success = true;
37 
38  if (extension == "bmp")
39  success = plot->saveBmp(filename, 0, 0, 1.0);
40  else if (extension == "pdf")
41  success = plot->savePdf(filename, false, 0, 0, "Vespucci 1.0", "Plot");
42  else if (extension == "png"){
43  bool ok;
44  int quality = QInputDialog::getInt(plot->parentWidget(), "Enter Quality",
45  "Quality (%)",

49  }
50 
51  else if (extension == "jpg"){
52  bool ok;
53  int quality = QInputDialog::getInt(plot->parentWidget(), "Enter Quality",
54  "Quality (%)",
55  80, 0, 100, 1, &ok);
56  if (ok)
57  plot->saveJpg(filename, 0, 0, 1.0, quality);
58  }
59  else if (extension == "svg"){
60  QPicture picture;
61  QCPPainter qcp_painter(&picture);
62  qcp_painter.setMode(QCPPainter::pmVectorized);
63

78  painter.end();

82  }

123  "Compression",
