52  QStringList command_list = macro.split("\n");
53  //this regular expression is used to split the commands so that the first
54  //member of the list is the function name, and all subsequent entries are
55  //parameters of the function.
56  QRegExp sep("(\(|\)|,)");
57  foreach (const QString &command, command_list){
58  QStringList pieces = command.split(sep);
59  QString function_name = pieces[0];
60  pieces.removeAt(0); //now pieces has only the parameters
61  QPair<QString, QStringList> processed_command(function_name, pieces);
62  commands_.push_back(processed_command);//insert at the end of the list
63  }
