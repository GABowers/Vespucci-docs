92 void DataExtractorDialog::on_buttonBox_accepted()
93 {
94  uvec indices;
95  QString name = name_line_edit_->text();
96  mat index_mat;
97 
98  if (method_combo_box_->currentText() == "Index File"){
99  string filename = filename_line_edit_->text().toStdString();
100  uword index;
101  index_mat.load(filename);
102  indices.set_size(index_mat.n_elem);
103  for (uword i = 0; i < index_mat.n_elem; ++i){
104  index = index_mat(i);
105  indices(i) = index;
106  }
107  }
108  else{
109  double upper = upper_box_->value();
110  double lower = lower_box_->value();
111  indices = arma::find(condition_ >= lower && condition_ <= upper);
112  }
113 
114  QFile *log_file = workspace->CreateLogFile(name);
115  QSharedPointer<VespucciDataset> new_dataset;
116  try{
117  new_dataset = QSharedPointer<VespucciDataset>(new VespucciDataset(name,
118  main_window_,
119  workspace->directory_ptr(),
120  log_file,
121  dataset_,
122  indices));
123  workspace->AddDataset(new_dataset);
124  }
125  catch(exception e){
126  main_window_->DisplayExceptionWarning(e);
127  }
128 
129  close();
130  dataset_.clear();
131 }
