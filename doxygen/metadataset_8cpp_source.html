110 mat MetaDataset::ProcessAverage(vec &x, vec &y)
111 {
112  mat spectra;
113  vec parent(1);
114  mat indices_buf(1, 3);
115  for (int i = 0; i < parent_datasets_.size(); ++i){
116  spectra.insert_cols(spectra.n_cols, parent_datasets_[i]->AverageSpectrum(false));
117  parent(0) = i + 1;
118  indices_buf(0, 0) = i + 1;
119  indices_buf(0, 1) = 0;
120  indices_buf(0, 2) = 0;
121  x.insert_rows(x.n_rows, parent);
122  y.insert_rows(y.n_rows, parent);
123  parent_coordinates_.insert_rows(parent_coordinates_.n_rows, indices_buf);
124  }
125  return spectra;
126 }
