30 {
31  QFile inputfile(filename);

34  QString line = inputstream.readLine();
35  QStringList tab_list = line.split("\t");
36  QStringList comma_list = line.split(",");
37  bool valid;
38 
39  if(tab_list.size() <= 0 && comma_list.size() <= 0){
40  return false;
41  }
42  else if(tab_list.size() > 0 && comma_list.size() > 0){
43  //we probably have a tab delimited file with commas for decimal points
44  //import function will probabably throw exception if comma is separator
45  //and type is wide text, so comma_decimals must be checked
46  comma_decimals = true;
47  return true;
48  }
49  else if(tab_list.size() > 0 && comma_list.size() == 0){
50  comma_decimals = false;
51  for (int i = 0; i < tab_list.size(); ++i){
52  tab_list[i].toDouble(&valid);
53  if(!valid)
54  return false;
55  }
56  return true;
57  }
58  else if(tab_list.size() == 0 && comma_list.size() > 0){
59  comma_decimals = false; //we don't know if this is true or not.
60  //users using instruments that save csv files with local number formatting
61  //in regions where commas are used for decimal places should be super
62  //careful.
63  for (int i = 0; i < comma_list.size(); ++i){
64  comma_list[i].toDouble(&valid);
65  if(!valid)
66  return false;
67  }
68  return true;
69  }
70  else{
71  return false;
72  }

98  std::vector<double> abs_s;

100  std::ifstream file_stream(filename);

102  std::getline(file_stream, current_line);
103  typedef boost::tokenizer<boost::char_separator<char> > tokenizer;
104  tokenizer::iterator it;
105  boost::char_separator<char> sep(",\t");
106  tokenizer tok(current_line, sep);
107  for (it = tok.begin(); it != tok.end(); ++it){
108  try{
109  abs_s.push_back(boost::lexical_cast<double>(*it));
110  }catch(boost::bad_lexical_cast & b){
111  std::cout << b.what() << std::endl;
112  }
113  }

115  std::getline(file_stream, current_line);
116  if (current_line.empty()){break;}
117  tok = tokenizer(current_line, sep);
118  it = tok.begin();
119  if (swap_spatial){
120  try{
121  y_s.push_back(boost::lexical_cast<double>(*it));
122  }catch (boost::bad_lexical_cast & b){
123  std::cout << b.what();
124  }
125 
126  ++it;

150  }

159  }
160  }

162  x = arma::vec(x_s);
163  y = arma::vec(y_s);
164  spectra = arma::mat(spec_s);
165  spectra.reshape(abscissa.n_rows, x.n_rows);
166 
167  //check to make sure everything is sorted the way Vespucci expects
168  arma::uvec sorted_indices = arma::stable_sort_index(abscissa);
169  abscissa = abscissa.rows(sorted_indices);
170  spectra = spectra.rows(sorted_indices);
171 
172  return true;
173 }

176 bool TextImport::ImportMultiplePoints(std::map<std::pair<int, int>, std::string> filenames,
177  int rows, int cols,
178  arma::mat &spectra,
179  arma::vec &abscissa,
180  arma::vec &x, arma::vec &y)
181 {
182  //fill x and y based on rows and colums
183  x.set_size(rows*cols);
184  y.set_size(rows*cols);
185 
186  bool have_abscissa = false;
187  mat current_spectrum;
188  std::pair<int,int> origin(0,0);
189  bool ok = current_spectrum.load(filenames[origin]);
190  int spec_rows = current_spectrum.n_rows;
191  spectra.set_size(spec_rows, x.n_rows);
192  if (!ok){
193  return false;
194  }
195  typedef std::map<std::pair<int,int>, std::string>::iterator map_it;
196  uword i = 0;
197  std::string filename;
198  for (map_it it = filenames.begin(); it!=filenames.end(); ++it){
199  std::pair<int,int> keys = it->first;
200  x(i) = (double) keys.first;
201  y(i) = (double) keys.second;
202  filename = it->second;
203  ok = current_spectrum.load(filename);
204  if (!have_abscissa && current_spectrum.n_cols == 2){
205  if (current_spectrum.n_rows > current_spectrum.n_cols){
206  abscissa = current_spectrum.col(0);
207  current_spectrum.shed_col(0);
208  }
209  else{
210  abscissa = current_spectrum.row(0).t();
211  current_spectrum.shed_row(0);
212  }
213  have_abscissa = true;
214  }
215  try{
216  if (current_spectrum.n_rows > current_spectrum.n_cols){
217  spectra.col(i) = current_spectrum.col(0);
218  }
219  else{
220  spectra.col(i) = current_spectrum.row(0).t();
221  }
222  }catch(std::exception e){
223  spectra.clear();
224  abscissa.clear();
225  x.clear();
226  y.clear();
227  std::cerr << "Exception thrown. Spectra " << i << "." << "Type: " << e.what() << "." << std::endl;
228  return false;
229  }
230 
231  if (!ok){
232  std::cerr << "Could not load file " << filename << "." << std::endl;
233  }
234  ++i;
235  }
236 
237  //check to make sure everything is sorted the way Vespucci expects
238  arma::uvec sorted_indices = arma::stable_sort_index(abscissa);
239  abscissa = abscissa.rows(sorted_indices);
240  spectra = spectra.rows(sorted_indices);
241 
242  return have_abscissa;
243 }

259 bool TextImport::ImportWitec(std::string spec_filename, std::string x_filename, std::string y_filename, arma::mat &spectra, arma::vec &abscissa, arma::vec &x, arma::vec &y)
260 {
261  bool spec_ok = spectra.load(spec_filename);
262  if (!spec_ok){
263  spectra.clear();
264  abscissa.clear();
265  x.clear();
266  y.clear();
267  return false;
268  }
269 
270  abscissa = spectra.col(0);
271  spectra.shed_col(0);
272  bool x_ok = x.load(x_filename);
273  bool y_ok = y.load(y_filename);
274  if (!(x_ok && y_ok)){
275  spectra.clear();
276  abscissa.clear();
277  x.clear();
278  y.clear();
279  return false;
280  }
281 
282  //the repeating, unique values in y are what we keep with constant x
283  //if y has the unique values, then we keep constant x
284  arma::vec unique_x = arma::unique(x);
285  arma::vec unique_y = arma::unique(y);
286  bool x_unique = unique_x.n_rows == x.n_rows;
287 
288  //if data doesn't look like we expect it to, give up
289  if (!x_unique && (unique_y.n_rows == y.n_rows)){
290  spectra.clear();
291  abscissa.clear();
292  x.clear();
293  y.clear();
294  return false;
295  }
296 
297 
298  if (x_unique){
299  //we're going to fill x with new values (y already follows Vespucci conventions)
300  x.clear();
301  for (arma::uword i = 0; i < unique_x.n_rows; ++i){
302  arma::vec x_values(unique_y.n_rows);
303  x_values.fill(unique_x(i));
304  x = arma::join_vert(x, x_values);
305  }
306  }
307  else{
308  //we're going to fill y with new values (x already follows Vespucci conventions)
309  y.clear();
310  for (arma::uword i = 0; i < unique_y.n_rows; ++i){
311  arma::vec y_values(unique_x.n_rows);
312  y_values.fill(unique_y(i));
313  y = arma::join_vert(y, y_values);
314  }
315  }
316 
317  return true;
318 }
