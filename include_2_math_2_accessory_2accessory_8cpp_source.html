135 arma::mat Vespucci::Math::spdiags(arma::mat B, arma::ivec d, arma::uword m, arma::uword n)
136 {
137  arma::uword i, j, k;
138  arma::uword size = B.n_cols;
139  arma::uword number = d.size();
140  arma::uword diag_size;
141  arma::uword column_size = B.n_rows;
142  arma::colvec column;
143  arma::mat output;
144  output.zeros();
145  output.set_size(m, n);
146  arma::colvec diagonal;
147  arma::ivec subdiagonals;
148  arma::ivec superdiagonals;
149 
150  if (number > size){
151  throw std::invalid_argument("spdiags: vector larger than diagonal of arma::matrix");
152  return output;
153  }
154 
155  superdiagonals = d(find (d >= 0));
156  subdiagonals = d(find (d < 0));
157 
158 
159 
160  for (i = 0; i < subdiagonals.size(); ++i){
161  k = subdiagonals(i);
162  diagonal = output.diag(k);
163  diag_size = diagonal.n_elem;
164  column = B.col(i);
165 
166  if (diag_size < column_size){
167  for (i = 0; i < diag_size; ++i){
168  output.diag(k)(i) = column(i);
169  }
170  }
171 
172  else {
173  for (i = 0; i < column_size; ++i){
174  output.diag(k)(i) = column(i);
175  }
176  }
177  }
178 
179  for (i = 0; i < superdiagonals.size(); ++i){
180  k = superdiagonals(i);
181  diag_size = output.diag(k).n_elem;
182  column = B.col(i);
183  if (diag_size < column_size){
184  j = column_size;
185  for (i = diag_size; i <=0; --i){
186  output.diag(k)(i) = column(j);
187  --j;
188  }
189  }
190 
191  else {
192  for (i = 0; i < column_size; ++i){
193  output.diag(k)(i) = column(i);
194  }
195  }
196  }
197  return output;
198 }

206 arma::mat Vespucci::Math::orth(arma::mat X)
207 {
208  double eps = arma::datum::eps;
209  arma::mat U, V;
210  arma::vec s;
211  svd(U, s, V, X);
212  double tol;
213  tol = std::max(X.n_rows*s(1)*eps, X.n_cols*s(1)*eps);
214  arma::uvec q1 = find(s > tol);
215  double rank = q1.n_elem;
216 
217  if (rank > 0){
218  return -1*U.cols(0, rank-1);
219  }
220  else{
221  std::cerr << "orth: no basis found" << std::endl;
222  //cout << "end orth" << endl;
223  return arma::zeros(X.n_rows, X.n_cols);

259 {

262  }

267  arma::vec padding;

468 void Vespucci::Math::position(arma::uword index, arma::uword n_rows, arma::uword n_cols, arma::uword &i, arma::uword &j)
469 {
470

472  //second row (1) is n to 2n - 1

483  j = (arma::uword) std::floor(index - i*n_cols);

502 double Vespucci::Math::quantile(arma::vec &data, double probs)
503 {
504  if (probs > 1 || probs < 0){
505  throw std::invalid_argument("quantile: probs must be between 0 and 1");
506  }
507  if (data.n_rows < 1){
508  throw std::invalid_argument("quantile: empty input vector");
509  }
510 
511  double h = (data.n_rows - 1)*probs; //normally you'd add one here, but remember indexing starts at 0
512 
513  if (data.in_range(data.in_range(std::floor(h) && std::floor(h) + 1))){
514  return data(std::floor(h)) + (h - std::floor(h))*(data(std::floor(h) + 1) - data(std::floor(h)));
515  }
516  //current or next value is too high, so take last
517  else if ((std::floor(h) >= data.n_rows) || (std::floor(h) + 1 >= data.n_rows)){
518  return data(data.n_rows - 1);
519  }
520  else{
521  throw std::runtime_error("quantile not computed");
522  }
523 }

526 double Vespucci::Math::mad(arma::vec &data)
527 {
528  double median;
529  double value;
530  arma::vec sorted = sort(data);
531  //calculate median (we do it this way because of an error in MinGW)
532  arma::uword center = sorted.n_rows / 2;
533  if (sorted.n_rows % 2 != 0){
534  median = 0.5*(sorted(center) + sorted(center+1));
535  }
536  else{
537  median = sorted(center);
538  }
539 
540  data -= median;
541  sorted = sort(data);
542 
543  if (sorted.n_rows % 2 != 0){
544  value = 0.5*(sorted(center) + sorted(center+1));
545  }
546  else{
547  value = sorted(center);
548  }
549  return value;
550 }

582 double Vespucci::Math::RecalculateStdDev(double new_value,
583  double old_mean,
584  double old_stddev,
585  double old_count)
586 
587 {
588  double sum_x = old_count*old_mean + new_value;
589  double sum_squares = old_count*(std::pow(old_stddev, 2.0)
590  + std::pow(old_mean, 2.0))
591  + std::pow(new_value, 2.0);

594  ((old_count + 1.0)*sum_squares - std::pow(sum_x, 2.0))
