402  settings_.setValue("ordLabel", "Intensity");

404  settings_.endGroup();
405  }
406  else{
407  settings_.beginGroup("specdata");

410  if (!settings_.allKeys().contains("absUnits"))
411  settings_.setValue("absUnits", "cm⁻¹");
