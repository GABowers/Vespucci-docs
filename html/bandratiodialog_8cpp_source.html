96 void BandRatioDialog::on_buttonBox_accepted()
97 {
98  if (first_min_line_edit_->text().isEmpty() || first_max_line_edit_->text().isEmpty() || second_min_line_edit_->text().isEmpty() || second_max_line_edit_->text().isEmpty()){
99  QMessageBox::warning(this, "Invalid Input!", "You must enter numbers for left and right bounds.");
100  return;
101  }
102 
103 
104  double first_entered_min = first_min_line_edit_->text().toDouble();
105  double second_entered_min = second_min_line_edit_->text().toDouble();
106  double first_entered_max = first_max_line_edit_->text().toDouble();
107  double second_entered_max = second_max_line_edit_->text().toDouble();
108 
109  if (first_entered_min < dataset_->abscissa_ptr()->min() || second_entered_min < dataset_->abscissa_ptr()->min()){
110  QMessageBox::warning(this, "Invalid Input!", "You have entered a left bound that is smaller than the smallest number on the spectral abscissa");
111  return;
112  }
113 
114  if (first_entered_max > dataset_->abscissa_ptr()->max() || second_entered_max > dataset_->abscissa_ptr()->max()){
115  QMessageBox::warning(this, "Invalid Input!", "You have entered a right bound that is larger than the largest number on the spectral abscissa");
116  return;
117  }
118  uint bound_window = search_window_spin_box_->value();
119 
120 
121  QString name = name_line_edit_->text();
122  QString value_method = method_combo_box_->currentText();
123 
124  if (value_method == "Riemann Sum"){
125  try{
126  dataset_->BandRatio(name, first_entered_min, first_entered_max,
127  second_entered_min, second_entered_max,
128  bound_window);
129  }catch(exception e){
130  workspace->main_window()->DisplayExceptionWarning(e);
131  }
132  }
133  else if (value_method == "Gaussian Fit"){
134  try{
135 
136  }catch(exception e){
137  workspace->main_window()->DisplayExceptionWarning(e);
138  }
139  }
140  else{
141  QMessageBox::warning(this, "Error Occurred", "A non-fatal error occurred: invalid input from method_combo_box_");
142  }
143 
144  close();
145  dataset_.clear();
146 }
