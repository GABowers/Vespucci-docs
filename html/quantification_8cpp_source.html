39 arma::rowvec Vespucci::Math::Quantification::QuantifyPeak(const arma::vec &spectrum, const arma::vec &abscissa, double &min, double &max, arma::uword bound_window, arma::mat &total_baseline, arma::mat &inflection_baseline)

41  //performs analysis of peak shape and magnitude
42  arma::rowvec results(8);

44  arma::uword min_index = Vespucci::Math::ClosestIndex(min, abscissa);
45  arma::uword max_index = Vespucci::Math::ClosestIndex(max, abscissa);
46 
47  min = abscissa(min_index);
48  max = abscissa(max_index);
49 
50  arma::uword min_start = min_index - bound_window;
51  arma::uword min_end = min_index + bound_window;
52  min_end = (min_end >= abscissa.n_rows ? abscissa.n_rows - 1 : min_end);
53 
54  arma::uword max_start = max_index - bound_window;
55  arma::uword max_end = max_index + bound_window;
56  max_end = (max_end >= abscissa.n_rows ? abscissa.n_rows - 1 : max_end);
57 
58  double total_area = arma::sum(spectrum.rows(min_index, max_index));
59  total_area = total_area / (abscissa(max_index) - abscissa(min_index));
60 
61  arma::uword window_size = max_index - min_index + 1;
62  total_baseline = arma::linspace(spectrum(min_index), spectrum(max_index), window_size);
63  arma::vec total_abscissa = abscissa.rows(min_index, max_index);
64  double baseline_area = arma::accu(total_baseline) / (abscissa(max_index) - abscissa(min_index));
65  total_baseline = arma::join_horiz(total_abscissa, total_baseline);
66 
67  arma::vec min_window = spectrum.rows(min_start, min_end);
68  arma::vec max_window = spectrum.rows(max_start, max_end);
69  arma::uword inflection_min_index = Vespucci::Math::LocalMinimum(min_window, min) + min_start;
70  arma::uword inflection_max_index = Vespucci::Math::LocalMinimum(max_window, max) + max_start;
71 
72  double area_between_inflection;
73  try{
74  area_between_inflection = arma::sum(spectrum.rows(min_index, max_index));
75  }catch(std::exception e){
76  std::cout << "Exception: invalid inflection points found" << std::endl;
77  area_between_inflection = std::nan("");
78  inflection_min_index = min_index;
79  inflection_max_index = max_index;
80  }
81 
82  area_between_inflection = area_between_inflection / (abscissa(inflection_max_index) - abscissa(inflection_min_index));
83  window_size = inflection_max_index - inflection_min_index + 1;
84  inflection_baseline = arma::linspace(spectrum(inflection_min_index), spectrum(inflection_max_index), window_size);
85  double inf_baseline_area = arma::accu(inflection_baseline) / (abscissa(inflection_min_index) - abscissa(inflection_max_index));
86 
87 
88 
89  //we need to make sure the peak center is within the originally specified range
90  arma::uword search_min_index = (inflection_min_index < min_index ? min_index : inflection_min_index);
91  arma::uword search_max_index = (inflection_max_index > max_index ? max_index : inflection_max_index);
92  arma::vec region = spectrum.rows(search_min_index, search_max_index);
93  arma::vec region_abscissa = abscissa.rows(search_min_index, search_max_index);
94  inflection_baseline = join_horiz(region_abscissa, region);
95  double maximum = region.max();
96  arma::uvec max_loc = arma::find(region == maximum);
97  arma::uword max_pos = max_loc(0);
98  arma::vec search_baseline = arma::linspace(region(0), region(region.n_rows - 1), region.n_rows);
99  double peak_center = region_abscissa(max_pos);
100  double adj_maximum = maximum - search_baseline(max_pos);
101 
102  double half_maximum = adj_maximum / 2.0;
103  //find left inflection points
104 
105  arma::vec bandwidth_region = spectrum.rows(search_min_index, search_max_index);
106  arma::vec bandwidth_baseline = arma::linspace(bandwidth_region(0), bandwidth_region(bandwidth_region.n_rows - 1), bandwidth_region.n_rows);
107  bandwidth_region = bandwidth_region - bandwidth_baseline;
108  arma::vec bandwidth_abscissa = abscissa.rows(search_min_index, search_max_index);
109  bandwidth_region = bandwidth_region - half_maximum * arma::ones(bandwidth_region.n_rows);
110  arma::uword i = 0;
111  //search for first positive point, point before is inflection
112  while (i < bandwidth_region.n_rows && bandwidth_region(i) < 0){++i;}
113  arma::uword left = (i >= bandwidth_region.n_rows ? bandwidth_region.n_rows - 1 : i);
114 
115  //search for first negative point after first infleciton
116  while (i < bandwidth_region.n_rows && bandwidth_region(i) >= 0){++i;}
117  arma::uword right = (i >= bandwidth_region.n_rows ? bandwidth_region.n_rows - 1 : i);
118 
119  if (bandwidth_region.n_rows > left)
120  left = std::fabs(bandwidth_region(left)) < std::fabs(bandwidth_region(left + 1)) ? left : left + 1;
121  if (right > 0)
122  right = std::fabs(bandwidth_region(right)) < std::fabs(bandwidth_region(right - 1)) ? right : right - 1;
123 
124 
125  double fwhm = bandwidth_abscissa(right) - bandwidth_abscissa(left);
126 
127  results(0) = peak_center;
128  results(1) = maximum;
129  results(2) = adj_maximum;
130  results(3) = total_area;
131  results(4) = total_area - baseline_area;
132  results(5) = area_between_inflection;
133  results(6) = area_between_inflection - inf_baseline_area;
134  results(7) = fwhm;
135 
136  return results;
137 }
