99 void BandRatioDialog::on_buttonBox_accepted()
100 {
101  if (first_min_box_->text().isEmpty() || first_max_box_->text().isEmpty() || second_min_box_->text().isEmpty() || second_max_box_->text().isEmpty()){
102  QMessageBox::warning(this, "Invalid Input!", "You must enter numbers for left and right bounds.");
103  return;
104  }
105 
106 
107  double first_entered_min = first_min_box_->text().toDouble();
108  double second_entered_min = second_min_box_->text().toDouble();
109  double first_entered_max = first_max_box_->text().toDouble();
110  double second_entered_max = second_max_box_->text().toDouble();
111 
112  if (first_entered_min < workspace->GetWavelengthMin(data_index_) || second_entered_min < workspace->GetWavelengthMin(data_index_)){
113  QMessageBox::warning(this, "Invalid Input!", "You have entered a left bound that is smaller than the smallest number on the spectral abscissa");
114  return;
115  }
116 
117  if (first_entered_max > workspace->GetWavelengthMax(data_index_) || second_entered_max > workspace->GetWavelengthMax(data_index_)){
118  QMessageBox::warning(this, "Invalid Input!", "You have entered a right bound that is larger than the largest number on the spectral abscissa");
119  return;
120  }
121 
122 
123 
124  QString name = name_box_->text();
125  QString value_method = value_method_selector_->currentText();
126  //QString integration_method = integration_method_selector_->currentText();
127 
128  UnivariateMethod::Method method;
129  if (value_method == "Area")
130  method = UnivariateMethod::AreaRatio;
131  else
132  method = UnivariateMethod::IntensityRatio;
133 
134  int gradient_index = color_selector_->currentIndex();
135  try{
136  data_->BandRatio(first_entered_min,
137  first_entered_max,
138  second_entered_min,
139  second_entered_max,
140  name,
141  method,
142  gradient_index);
143  }catch(exception e){
144  workspace->main_window()->DisplayExceptionWarning(e);
145  }
146 
147  this->close();
148  data_.clear();
149 }
