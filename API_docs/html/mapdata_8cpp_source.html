512 void MapData::DrawScaleBar(double width,
513  double height,
514  QString units,
515  QColor color,
516  QString position,
517  QFont font)
518 {
519  //instantiate shape
520 
521  //QCustomPlot object will take ownership of these and make sure they're deleted.
522  QCPItemRect *scale_bar = new QCPItemRect(map_qcp_);
523  QCPItemText *scale_bar_text = new QCPItemText(map_qcp_);
524 
525 
526  map_qcp_->addItem(scale_bar);
527  map_qcp_->addItem(scale_bar_text);
528 
529  QString text = QString::number(width) + " " + units;
530  scale_bar_text->setText(text);
531  scale_bar_text->setPositionAlignment(Qt::AlignCenter);
532  scale_bar_text->setColor(color);
533  scale_bar_text->setFont(font);
534 
535  QPen pen(color);
536  QBrush brush(color, Qt::SolidPattern);
537  scale_bar->setPen(pen);
538  scale_bar->setBrush(brush);
539  QCPRange key_range = map_->keyAxis()->range();
540  QCPRange value_range = map_->valueAxis()->range();
541 
542  //find corners of image
543  double key_max = key_range.upper;
544  double key_min = key_range.lower;
545  double value_max = value_range.upper;
546  double value_min = value_range.lower;
547 
548  double x_min, y_min, x_max, y_max;
549 
550  if (position == "Top Left"){
551  x_min = key_min + 2.0;
552  x_max = x_min + width;
553  y_max = value_max - 4.0;
554  y_min = y_max - height;
555  }
556 
557  else if (position == "Top Right"){
558  x_max = key_max - 2.0;
559  x_min = x_max - width;
560  y_max = value_max - 4.0;
561  y_min = y_max - height;
562  }
563 
564  else if (position == "Bottom Left"){
565  x_min = key_min + 2.0;
566  x_max = x_min + width;
567  y_min = value_min + 2.0;
568  y_max = y_min + height;
569  }
570 
571  else{
572  x_max = key_max - 2.0;
573  x_min = x_max - width;
574  y_min = value_min + 2.0;
575  y_max = y_min + height;
576  }
577 
578  scale_bar->bottomRight->setCoords(x_max, y_min);
579  scale_bar->topLeft->setCoords(x_min, y_max);
580  double x_mid = ((x_max - x_min) / 2.0) + x_min;
581  double y_text = y_max + 1;
582  if (font.pointSize() >= 10){
583  y_text += (font.pointSize() - 10) / 4;
584  }
585 
586  scale_bar_text->position->setCoords(x_mid, y_text);
587 
588  map_qcp_->show();
589 }
