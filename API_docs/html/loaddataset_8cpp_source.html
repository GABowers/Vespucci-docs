83 void LoadDataset::on_buttonBox_accepted()
84 {
85  QString y_description = y_description_box_->text() + " (" + y_units_box_->text() + ")";
86  QString x_description = x_description_box_->text() + " (" + x_units_box_->text() + ")";
87 
88  QString name = name_box_->text();
89  QString filename = filename_line_edit_->text();
90  QFileInfo file_info(filename);
91  QString extension = file_info.suffix();
92 
93  std::string format;
94  const QString data_format_string = data_format_box_->currentText();
95  if (extension == "vds" || (data_format_string == "Vespucci Dataset" && extension != "txt"))
96  format = "VespucciBinary";
97  else if (data_format_string == "Wide Text")
98  format = "WideTabDel";
99  else if (data_format_string == "Wide Text (CSV)")
100  format = "WideCSV";
101  else if (data_format_string == "Long Text")
102  format = "LongTabDel";
103  else if (data_format_string == "Long Text (CSV)")
104  format = "LongCSV";
105  else
106  return;
107 
108 
109 
110  //bool remove_at_position = false;
111  //int temp_position=0;
112  //int position;
113  bool swap = swap_check_box_->isChecked();
114 
115  QStringList names_list = workspace->dataset_names();
116  QStringList::iterator i;
117 
118  int warning_response=QMessageBox::Yes;
119  //Warning message pops up when user enters a name that already exists
120  for (i=names_list.begin(); i != names_list.end(); ++i){
121  if (name==(*i)){
122  warning_response =
123  QMessageBox::question(this,
124  "Duplicate Name",
125  "There is already a dataset with this"
126  " name in the workspace. Datasets are"
127  " not indexed by name, but using two"
128  " datasets with the same name may lead"
129  " to confusion. Are you sure you wish"
130  " to continue with this name?",
131  QMessageBox::No,
132  QMessageBox::Yes);
133  }
134  }
135 
136 
137  if (warning_response == QMessageBox::Yes && file_info.exists()){
138  QFile *log_file = workspace->CreateLogFile(name);
139 
140  if (format == "VespucciBinary"){
141  try{
142  QSharedPointer<VespucciDataset> dataset(new VespucciDataset(filename,
143  workspace->main_window(),
144  directory_,
145  log_file,
146  name,
147  x_description,
148  y_description));
149  workspace->AddDataset(dataset);
150  workspace->set_directory(file_info.dir().absolutePath());
151  //dataset.clear(); //dataset should be copied to list
152  }
153  catch(exception e){
154  workspace->main_window()->DisplayExceptionWarning(e);
155  delete log_file;
156  return;
157  }
158  }
159 
160  else{
161  try{
162  QSharedPointer<VespucciDataset> dataset(new VespucciDataset(filename,
163  workspace->main_window(),
164  directory_,
165  log_file,
166  name,
167  x_description,
168  y_description,
169  swap,
170  format));
171  if (!dataset->ConstructorCancelled()){
172  workspace->AddDataset(dataset);
173  workspace->set_directory(file_info.dir().absolutePath());
174  dataset.clear();
175  }
176  }
177  catch(exception e){
178  workspace->main_window()->DisplayExceptionWarning(e);
179  delete log_file;
180  return;
181  }
182  }
183 
184  }
185  this->close();
186 
187 }
