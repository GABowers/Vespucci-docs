29 void Vespucci::Math::Fitting::Gaussian(double *p, double *x, int m, int n, void *data)
30 {
31  //Gaussian function:
32  //y = y0 + A/(w*sqrt(PI/(4*ln(2)))) * exp(-4*ln(2)*(x-xc)^2/w^2)
33  //p[0] = y0
34  //p[1] = xc
35  //p[2] = A
36  //p[3] = w
37 
38 
39 
40  int i;
41  for (i = 0; i < n; ++i){
42  x[i] = p[0] //y axis offset
43  + (p[2]/(p[3]*std::sqrt(arma::datum::pi/(4*std::log(2)))) ) //"amplitude" term
44  * std::exp(-4*std::log(2)*std::pow((p[1] - (double) i), 2.0) / std::pow(p[3], 2.0)); //exponential "width" term
45  }

50 void Vespucci::Math::Fitting::GaussianJacobian(double *p, double *jac, int m, int n, void *data)
51 {
52  int i, j;
53  for(i=j=0; i<n; ++i){
54 
55  //with respect to y0
56  jac[j++] = 1.0; //the only term for y0 is itself and a constant
57 
58  //with respect to xc
59  jac[j++] = (std::pow(i, 2.0) / std::pow(p[3], 2))*-4*std::log(2) //chain rule term for xc
60  *(p[2]/(p[3]*std::sqrt(arma::datum::pi/(4*std::log(2)))) ) //"amplitude" term
61  * std::exp(-4*std::log(2)*std::pow((p[1] - (double) i), 2.0) / std::pow(p[3], 2.0)); //exponential "width" term
62 
63  //with respect to A
64  jac[j++] = (1.0 /(p[3]*std::sqrt(arma::datum::pi/(4*std::log(2)))) ) //"amplitude" term without A
65  * std::exp(-4*std::log(2)*std::pow((p[1] - (double) i), 2.0) / std::pow(p[3], 2.0)); //exponential "width" term
66 
67  //with respect to w
68  jac[j++] = ( (2.0 * (-4*std::log(2)*std::pow((p[1] - (double) i), 2.0) / std::pow(p[3], 2.0)) + std::pow(p[3], 2.0)) / std::pow(p[3], 4.0)) //chain rule and product rule stuff
69  * (p[2]/(std::sqrt(arma::datum::pi/(4*std::log(2)))) ) //"amplitude" term without w
70  * std::exp(-4*std::log(2)*std::pow((p[1] - (double) i), 2.0) / std::pow(p[3], 2.0)); //exponential "width" term
71  }
72 }
