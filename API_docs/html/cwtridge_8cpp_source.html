129 double Vespucci::Math::CWTRidge::EstimateSNR(std::string method, arma::uword window_size, const arma::vec &noise)
130 {
131  double signal = coefs_.max();
132  arma::uword start, end, center;
133  try{
134  center = this->PeakCenter();
135  }
136  catch(std::exception e){
137  std::cerr << "SNR(): error finding peak center" << std::endl;
138  throw e;
139  }
140  start = (center > window_size ? center - window_size : 0);
141  end = (center + window_size >= coefs_.n_elem ? noise.n_elem - 1 : center + window_size);
142  arma::vec abs_noise;
143  try{
144  abs_noise = arma::abs(noise.rows(start, end)); //first scale must be sent
145  }
146  catch(std::exception e){
147  std::cerr << "SNR(): error calculating noise" << std::endl;
148  std::cerr << "center = " << center << std::endl;
149  std::cerr << "start = " << start << "\t end = " << end << std::endl;
150  std::cerr << "noise has " << noise.n_rows << " rows" << std::endl;
151  throw e;
152  }
153 
154  double noise_level;
155  if (method == "quantile"){
156  noise_level = Vespucci::Math::quantile(abs_noise, 0.95);
157  }
158  else if (method == "mean"){
159  noise_level = arma::mean(abs_noise);
160  }
161  else if (method == "stddev"){
162  noise_level = arma::stddev(abs_noise);
163  }
164  else{
165  throw std::invalid_argument("CWTRidge::SNR: " + method + " not a valid SNR method");
166  }
167  calculated_snr_ = std::abs(signal / noise_level);
168  return calculated_snr_;
169 }

231  arma::uword scl = scale();

234  arma::uword last = ((center + 3*scl) < spectra.n_rows ? center + 3*scl : spectra.n_rows - 1);
235  arma::vec X;
236  try{
237  X = spectra.rows(first, last);//search

242  std::cerr << "last = " << last << std::endl;

244  std::cerr << "error resizing X" <<std::endl;

246  }
247

254  throw e;
255  }
256  try{
257  d2X = Vespucci::Math::Transform::cwt(dX.col(0), "haar", scales);
258  }catch(std::exception e){
259  std::cerr << "error calculating d2X!" << std::endl;
260  throw e;

267  throw e;
268  }
269 
270  //find the local minima larger than and smaller than the center index

275

294  }

296  }catch(std::exception e){
