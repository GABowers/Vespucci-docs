180 bool VespucciWorkspace::RecalculateGlobalDataRange(QCPRange *new_data_range)
181 {
182  if (new_data_range->upper > global_data_range_.upper){
183  global_data_range_.upper = new_data_range->upper;
184  return true;
185  }
186  if (new_data_range->lower < global_data_range_.lower){
187  global_data_range_.lower = new_data_range->lower;
188  return true;
189  }
190  return false;
191 }
