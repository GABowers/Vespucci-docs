34 bool Vespucci::Math::DimensionReduction::VCA(const arma::mat &R, arma::uword p,
35  arma::uvec &indices, arma::mat &endmember_spectra,
36  arma::mat &projected_data, arma::mat &fractional_abundances)
37 {
38 //Initializations
39  arma::uword L = R.n_rows;
40  arma::uword N = R.n_cols;
41  if (L == 0 || N == 0){
42  std::cerr << "No data!" << std::endl;
43  return false;
44  }
45 
46  if (p > L){
47  std::cerr << "wrong number of endmembers (" << p << ")!"<< std::endl;
48  std::cerr << "set to 5 or one less than number of spectra" << std::endl;
49  p = (L < 5? 5: L-1);
50  }
51 //mat::mations of SNR
52  arma::mat r_m = mean(R, 1);
53  arma::mat R_m = arma::repmat(r_m, 1, N); //the mean of each spectral band
54  arma::mat R_o = R - R_m; //mean-center the data
55  arma::mat Ud;
56  arma::vec Sd;
57  arma::mat Vd;
58  arma::svds(Ud, Sd, Vd, arma::sp_mat(arma::cov(R_o)/N), p);
59  arma::mat x_p = trans(Ud) * R_o;
60  double SNR = Vespucci::Math::DimensionReduction::estimate_snr(R, r_m, x_p);
61  double SNR_th = 15 + 10*log10(p);
62 
63 //Choose projective projection or projection to p-1 subspace
64  arma::mat y;
65  if (SNR < SNR_th){
66  arma::uword d = p - 1;
67  Ud = Ud.cols(0, d-1);
68  projected_data = Ud * x_p.rows(0, d-1) + R_m; //in dimension L
69  arma::mat x = x_p.rows(0, d-1);//x_p = trans(Ud)*R_o, p-dimensional subspace
70  //following three lines are one in arma::matlab...
71  arma::mat sum_squares = sum(pow(x, 2));
72  double c = sum_squares.max();
73  c = std::sqrt(c);
74  y = join_vert(x, c*arma::ones(1, N));
75  }
76  else{
77  arma::uword d = p;
78  arma::svds(Ud, Sd, Vd, arma::sp_mat(arma::cov(R)/N), d);//R_o is a mean centered version...
79  x_p = trans(Ud)*R;
80  projected_data = Ud * x_p.rows(0, d-1);
81  arma::mat x = trans(Ud) * R;
82  arma::mat u = mean(x, 1);
83  y = x / arma::repmat(sum(x % arma::repmat(u, 1, N)), d, 1);
84  }
85 
86 
87  // The VCA algorithm
88  arma::vec w;
89  w.set_size(p);
90  arma::vec f;
91  arma::rowvec v;
92  indices.set_size(p);
93  //there are no fill functions for arma::uvecs
94  for (arma::uword i = 0; i < p; ++i)
95  indices(i) = 0;
96 
97  arma::mat A = arma::zeros(p, p);
98  double v_max;
99  double sum_squares;
100  arma::uvec q1;
101  A(p-1, 0) = 1;
102  for (arma::uword i = 0; i < p; ++i){
103  w.randu();
104  f = w - A*pinv(A)*w;
105  sum_squares = sqrt(sum(square(f)));
106  f /= sum_squares;
107  v = trans(f) * y;
108  v_max = max(abs(v));
109  q1 = arma::find(abs(v) == v_max, 1);
110  indices(i) = q1(0);
111  A.col(i) = y.col(indices(i)); //same as x.col(indices(i));
112  }
113  endmember_spectra = projected_data.cols(indices);
114  fractional_abundances = trans(pinv(endmember_spectra) * projected_data);
115  return true;
116 }

126 double Vespucci::Math::DimensionReduction::estimate_snr(const arma::mat &R, arma::vec r_m, arma::mat x)
127 {
128  arma::uword L = R.n_rows;
129  arma::uword N = R.n_cols;
130  arma::uword p = x.n_rows;
131  if (x.n_cols != N){
132  std::cerr << "invaliid input to Estimate_snr" << std::endl;
133  return 0;
134  }
135 
136  double P_y = accu(arma::square(R))/N;
137  double P_x = accu(arma::square(x))/N + dot(r_m, r_m);
138  return 10*log10( (P_x - p/L*P_y)/(P_y - P_x) );
139 }

149 int Vespucci::Math::DimensionReduction::HySime(arma::mat y,
150  arma::mat n,
151  arma::mat Rn,
152  arma::mat &Ek)
153 {
154  std::cout << "Vespucci::Math::DimensionReduction::HySime()" << std::endl;
155  if (n.n_rows != y.n_rows || n.n_cols != y.n_cols){
156  std::cerr << "HySime: Empty noise arma::matrix or its size does not agree with size of y" << std::endl;
157  throw(std::runtime_error("HySime: Empty noise arma::matrix or does not agree with size of y"));
158  }
159 
160  if (Rn.n_rows != Rn.n_cols || Rn.n_rows != y.n_rows){
161  std::cerr << "Bad noise correlation arma::matrix" << std::endl;
162  Rn = n*n.t() / y.n_cols;
163  }
164 
165  arma::mat x = y - n;
166 
167  arma::mat Ry = y*y.t() / y.n_cols;
168  arma::mat Rx = x*x.t() / y.n_cols;
169  arma::mat E, D, V;
170  arma::vec dx;
171  svd(E, dx, V, Rx);
172  D = arma::diagmat(dx);
173 
174  Rn = Rn + sum(Rx.diag())/y.n_rows/10000*arma::eye(y.n_rows, y.n_rows);
175 
176  arma::mat P = E.t() * Ry * E;
177  arma::vec Py = P.diag();
178 
179  P = E.t() * Rn * E;
180  arma::vec Pn = P.diag();
181 
182  arma::vec cost_F = -Py + 2 * Pn;
183  arma::uvec negatives = find (cost_F < 0);
184  int kf = negatives.n_elem;
185  arma::uvec sorted_cols = sort_index(cost_F);
186  Ek = E.cols(sorted_cols);
187  return kf;
188 
189 }

198 void Vespucci::Math::DimensionReduction::EstimateAdditiveNoise(arma::mat &noise, arma::mat &noise_correlation, const arma::mat &sample)
199 {
200  double small = 1e-6;
201  noise = arma::zeros(sample.n_rows, sample.n_cols);
202  arma::mat RR = sample * sample.t();
203  arma::mat RRi = arma::inv(RR + small*arma::eye(sample.n_rows, sample.n_rows));
204  arma::mat XX, RRa, beta;
205  for (arma::uword i = 0; i < sample.n_rows; ++i){
206  XX = RRi - (RRi.col(i) * RRi.row(i))/RRi(i, i);
207  RRa = RR.col(i);
208  RRa(i) = 0;
209  beta = XX * RRa;
210  beta(i) = 0;
211  noise.row(i) = sample.row(i) - beta.t() * sample;
212  }
213  arma::mat nn = noise * noise.t() / sample.n_rows;
214  noise_correlation = arma::diagmat(nn.diag());
215 }
