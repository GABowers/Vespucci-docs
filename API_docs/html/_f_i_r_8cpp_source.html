37 arma::mat Vespucci::Math::Smoothing::sgolay(arma::uword poly_order,
38  arma::uword window_size,
39  arma::uword deriv_order,
40  arma::uword scaling_factor)
41 {
42  if (window_size % 2 == 0){
43  std::cerr << "sgolay: invalid window size. Window size must be odd." << std::endl;
44  std::cerr << "using window size poly_order + 2 or nearest odd value" << std::endl;
45  window_size = poly_order + 3 - (poly_order % 2);
46  }
47 
48  if (deriv_order > poly_order){
49  std::cerr << "sgolay: derivative order greater than polynomial order." << std::endl;
50  std::cerr << "using derivative order poly order - 1" << std::endl;
51  deriv_order = poly_order - 1;
52  }
53  arma::mat F = arma::zeros(window_size, window_size);
54  arma::uword k = (window_size - 1) / 2;
55  arma::mat C(window_size, poly_order + 1);
56  arma::sword signed_window_size = window_size;
57  arma::vec column;
58  arma::sword row; //starting at 1 not 0
59  arma::mat A;
60  for (arma::uword i = 0; i <= k; ++i){
61  row = i + 1;
62  column = arma::linspace<arma::vec>((1-row), (signed_window_size - row), window_size);
63  for (arma::uword j = 0; j <= poly_order; ++j){
64  C.col(j) = arma::pow(column, j);
65  }
66  A = pinv(C);
67  F.row(i) = A.row(deriv_order);
68  }
69  arma::sword sign = (deriv_order % 2 == 0 ? 1: -1);
70  arma::uword start_index = 0;
71  for (arma::uword i = window_size - 1; i >k; --i){
72  F.row(i) = sign*F.row(start_index);
73  start_index++;
74  }
75  double product = (deriv_order == 0 ? 1 : prod(arma::linspace<arma::vec>(1, deriv_order, deriv_order)));
76  double power = pow(scaling_factor, deriv_order);
77  F /= (power/product);
78  return F;
79 }

127 arma::vec Vespucci::Math::Smoothing::ApplyFilter(const arma::vec &x, arma::mat coefficients, arma::uword window_size)
128 {
129  arma::uword k = (window_size - 1) / 2;
130  arma::vec filter = trans(coefficients.row(k));
131  //coefficients output by sgolay are in reverse order expected.
132  filter = fliplr(filter);
133  arma::vec filtered_data = conv(filter, x);
134  //for conv() to be equivalent to filter, we have to chop off the last window_size - 1 elements
135  filtered_data = filtered_data.rows(0, x.n_rows - 1);
136  //Now we have to fix the ends of the filtered data
137  filtered_data.rows(0, k) = coefficients.rows(k, window_size-1)*x.rows(0, window_size-1);
138  filtered_data.rows(x.n_rows - k, x.n_rows - 1) = coefficients.rows(0, k - 1)*x.rows(x.n_rows - window_size, x.n_rows - 1);
139  return filtered_data;
140 }
