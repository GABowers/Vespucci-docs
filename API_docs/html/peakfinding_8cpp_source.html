45 arma::vec Vespucci::Math::PeakFinding::FindPeaks(arma::vec X,
46  arma::vec dX,
47  double sel,
48  double threshold,
49  arma::vec &peak_magnitudes)
50 {
51  arma::uvec peak_indices;
52  arma::uvec peak_locations;
53  peak_magnitudes = arma::zeros(X.n_rows, 1);
54 
55  dX.elem( find(dX == 0) ).fill(-arma::datum::eps); //to find first of repeated values
56  double minimum_magnitude = X.min();
57  double left_min = X(0);
58  arma::uword length = X.n_elem;
59  arma::uword temporary_location = 0;
60 
61  //Find where derivative changes sign:
62  arma::uvec extrema_indices = find( (dX.subvec(0, length - 2) % dX.subvec(1, length - 1) )< 0);
63  arma::vec X_extrema = X.elem(extrema_indices);
64  length = X_extrema.n_elem;
65  double temporary_magnitude = minimum_magnitude;
66  bool peak_found = false;
67  arma::uword ii;
68  if (X(1) >= X(2))
69  ii = 1;
70  else
71  ii = 2;
72 
73  //This loop finds the peak locations
74  while (ii < length){
75  //reset peak finding if we had a peak earlier and the next peak is
76  //bigger than the last or the left minimum was small enough to reset
77  if (peak_found){
78  temporary_magnitude = minimum_magnitude;
79  peak_found = false;
80  }
81 
82  if ( (X_extrema(ii) > temporary_magnitude) && (X_extrema(ii) > left_min + sel) ){
83  std::cout << "temporary_location = " << ii << std::endl;
84  temporary_magnitude = X_extrema(ii);
85  }
86 
87  ++ii; //move into the valley
88  if (ii >= length){
89  break;
90  }
91  //come down at least sel from peak
92  if (!peak_found && (temporary_magnitude > sel + X_extrema(ii) ) ){
93  peak_found = true;
94  std::cout << "peak found!" << std::endl;
95  left_min = X_extrema(ii);
96  peak_locations << temporary_location;
97  }
98  else if(X(ii) < left_min)
99  left_min = X_extrema(ii);
100 
101  ++ii;
102 
103 
104  }
105  //remove all peaks below the threshold
106 
107  arma::vec results = arma::zeros(X.n_elem, 1);
108  peak_indices = extrema_indices.elem(peak_locations);
109  results.elem(peak_indices).ones();
110  peak_magnitudes.elem(peak_indices) = X(peak_indices);
111 
112 
113 
114  arma::uvec indices = find(peak_magnitudes <= threshold);
115  peak_magnitudes.elem(indices).zeros();
116  results.elem(indices).zeros();
117  return results;
118 }

132 arma::mat Vespucci::Math::PeakFinding::FindPeaksMat(arma::mat X, double sel, double threshold, arma::uword poly_order, arma::uword window_size, arma::mat &peak_magnitudes)
133 {
134  arma::mat peak_locations(X.n_rows, X.n_cols); //value that gets returned
135  peak_magnitudes.set_size(X.n_rows, X.n_cols);
136 
137  //calculate a smoothed first derivative:
138  arma::mat derivatized = Vespucci::Math::Smoothing::sgolayfilt(X, poly_order, window_size, 1, 1);
139  arma::vec thresh(X.n_cols);
140  arma::vec sels(X.n_cols);
141  if (std::isnan(sel)){
142  for (arma::uword i = 0; i < X.n_cols; ++i){
143  sels(i) = (X.col(i).max() - X.col(i).min() ) / 16.0;
144  }
145  }
146  else{
147  sels.fill(sel);
148  }
149 
150  if (std::isnan(threshold)){
151  arma::uword k = (X.n_rows % 2 == 0 ? (X.n_rows / 2) : (X.n_rows + 1 / 2 ));
152  arma::vec buffer(X.n_rows);
153  arma::uvec sorted;
154  //we calculate median this way due to a problem with my MinGW compiler that
155  //causes crashes when arma::median is called.
156  for (arma::uword i = 0; i < X.n_cols; ++i){
157  buffer = Vespucci::Math::diff(X, 1); //use arithmetic derivative, rather than smoothed derivative
158  sorted = stable_sort_index(buffer);
159  thresh(i) = buffer(sorted(k));
160  }
161  }
162  else{
163  thresh.fill(threshold);
164  }
165  arma::vec current_spectrum;
166  arma::vec current_derivative;
167  arma::vec current_magnitudes;
168  arma::vec current_peaks;
169  for (arma::uword i = 0; i < X.n_cols; ++i){
170  current_spectrum = X.col(i);
171  current_derivative = derivatized.col(i);
172  current_peaks = Vespucci::Math::PeakFinding::FindPeaks(current_spectrum, current_derivative, sels(i), thresh(i), current_magnitudes);
173  peak_locations.col(i) = current_peaks;
174  peak_magnitudes.col(i) = current_magnitudes;
175  }
176  return peak_locations;
177 }

202 arma::umat Vespucci::Math::PeakFinding::FindPeakPositions(arma::vec X, arma::vec dX,
203  double threshold,
204  std::string threshold_method,
205  arma::vec &peak_magnitudes)
206 {
207  //threshold can be arbitrary or calculated
208 
209  arma::vec d2X = Vespucci::Math::diff(dX, 1);
210  try{
211  d2X.insert_rows(0, 1, true);
212  }catch(std::exception e){
213  std::cerr << "error in Vespucci::Math::PeakFinding::FindPeakPositions" << std::endl;
214  }
215 
216  if (threshold_method == "count" && threshold <= 0){
217  throw std::runtime_error("Invalid threshold value for given method");
218  }
219  if ((threshold_method == "countpercentage" || threshold_method == "ratio") && threshold > 1){
220  throw std::runtime_error("Invalid threshold value for given method");
221  }
222 
223  //find where first derivative changes sign
224  arma::uvec extrema_indices = find( (dX.subvec(0, X.n_rows - 2) % dX.subvec(1, X.n_rows - 1) )< 0);
225  arma::vec d2X_extrema = d2X(extrema_indices);
226  arma::uvec maxima_indices = find(d2X_extrema < 0); //the indices in d2X_extrema, X_extrema and extrema_indices that correspond to maxima
227  arma::umat results(maxima_indices.n_elem, 3);
228  arma::uvec buffer(3);
229  peak_magnitudes.set_size(maxima_indices.n_elem); //each maximum corresponds to a magnitude
230 
231  //we have the indices of the maxima, now we need to find magnitudes and positions
232  //of the minima associated with each maximum
233  arma::uword center_index = 0;
234  arma::uword left_index = 0;
235  arma::uword right_index = 0;
236 
237  //find peak edges for each center and the magnitude of the peaks
238  arma::uword i;
239  try{
240  for (i = 0; i < maxima_indices.n_elem; ++i){
241 
242  if(maxima_indices(i) == 0){
243  //no left index (at beginning)
244  center_index = extrema_indices(maxima_indices(i)); //index of the center of peak
245  right_index = extrema_indices(maxima_indices(i) + 1); //index of right edge
246  left_index = center_index; //left edge is same as center
247  }
248 
249  else if(maxima_indices(i) + 1 >= extrema_indices.n_elem){
250  //no right_index (we're at the end)
251  center_index = extrema_indices(maxima_indices(i)); //index of the center of peak
252  left_index = extrema_indices(maxima_indices(i) - 1); //index of left edge
253  right_index = center_index;
254  }
255 
256  else{
257  //normal
258  center_index = extrema_indices(maxima_indices(i)); //index of the center of peak
259  left_index = extrema_indices(maxima_indices(i) - 1); //index of left edge
260  right_index = extrema_indices(maxima_indices(i) + 1); //index of right edge
261  }
262 
263  //if left_index == right_index, average is the same as difference...
264  peak_magnitudes(i) = (2.0*X(center_index) - X(left_index) - X(right_index))/2.0;
265  buffer(0) = center_index;
266  buffer(1) = left_index;
267  buffer(2) = right_index;
268  results.row(i) = buffer.t();
269  }
270  }catch(std::exception e){
271  std::cout << "Exception!" << std::endl;
272  std::cout << e.what() << std::endl;
273  std::cout << "iterator index = " << i << std::endl;
274  std::cout << "maxima index = " << maxima_indices(i) << std::endl;
275  std::cout << "extrema_indices.n_rows = " << extrema_indices.n_rows << std::endl;
276  std::cout << "center_index = " << center_index << std::endl;
277  std::cout << "left_index = " << left_index << std::endl;
278  std::cout << "right_index = " << right_index << std::endl;
279  throw std::runtime_error(e.what());
280  }
281 
282  //Fix thresholds//
283  //sort with largest first:
284  arma::uvec sorted_peak_indices = stable_sort_index(peak_magnitudes, "descend");
285  peak_magnitudes = peak_magnitudes.rows(sorted_peak_indices);
286  results = results.rows(sorted_peak_indices);
287 
288  if (threshold_method == "magnitude"){
289  arma::uvec indices = find(peak_magnitudes > threshold);
290  peak_magnitudes = peak_magnitudes.rows(indices);
291  return results.rows(indices);
292  }
293  else if (threshold_method == "count"){
294  arma::uword uthresh = (std::ceil(threshold) < results.n_rows - 1 ? std::ceil(threshold) : results.n_rows - 1);
295  peak_magnitudes = peak_magnitudes.rows(0, uthresh);
296  return results.rows(0, uthresh);
297  }
298  else if (threshold_method == "ratio"){
299  if (threshold == 1){
300  peak_magnitudes = peak_magnitudes.row(0);
301  return results.row(0);
302  }
303  else{
304  double cutoff = peak_magnitudes(0) * threshold;
305  arma::uvec indices = find(peak_magnitudes > cutoff);
306  peak_magnitudes = peak_magnitudes.rows(indices);
307  return results.rows(indices);
308  }
309  }
310  else{ //(threshold_method == "countpercentage")
311  arma::uword count = std::floor(threshold*results.n_rows);
312  peak_magnitudes = peak_magnitudes.rows(0, count);
313  return results.rows(0, count);
314  }

352 arma::vec Vespucci::Math::PeakFinding::EstimateBaseline(arma::vec X,
353  arma::umat peaks,
354  arma::uword window_size)
355 {
356  if (window_size % 2 == 0){
357  window_size--;
358  std::cerr << "invalid window_size, using one less" << std::endl;
359  }
360 
361  arma::vec baseline = X;
362  arma::uword start, end, center, size;
363  //linearize across bases of peaks
364  arma::uword i;
365  try{
366  for (i = 0; i < peaks.n_rows; ++i){
367  center = peaks(i, 0);
368  start = peaks(i, 1);
369  end = peaks(i, 2);
370  //cut accross in a straight line when peak is on edge of spectrum
371  size = end - start;
372 
373  //linearize baseline across base of peak
374  // if the start or end of the peak is the center, take a flat line
375  // from the minimum
376  if (start == center){
377  baseline.subvec(start, end-1) = X(end) * arma::ones(size);
378  }
379  else if (end == center){
380  baseline.subvec(start, end-1) = X(start) * arma::ones(size);
381  }
382  else{
383  baseline.subvec(start, end-1) = arma::linspace(X(start), X(end), size);
384  }
385 
386  }
387  }catch(std::exception e){
388  std::cout << "Exception! (peak exclusion)" << std::endl;
389  std::cout << "size of peaks = " << peaks.n_rows << " " << peaks.n_cols << std::endl;
390  std::cout << "i = " << i << std::endl;
391  throw std::runtime_error(e.what());
392  }
393 
394  //apply local minimum filtering to baseline
395 
396  arma::uword k = (window_size - 1) / 2; //mid-point of window
397  arma::vec buffer;
398  arma::vec filtered(baseline.n_elem);
399  try{
400  for (arma::uword i = k; i < (X.n_rows - k); ++i){
401  buffer = baseline.subvec(i-k, i+k);
402  filtered(i) = buffer.min();
403  }
404  //fill edges with first and last values
405  for (arma::uword i = 0; i < k; ++i){
406  filtered(i) = filtered(k);
407  }
408  for (arma::uword i = filtered.n_rows - k; i < filtered.n_elem; ++i){
409  filtered(i) = filtered(filtered.n_rows - k - 2);
410  }
411  }catch(std::exception e){
412  std::cout << "Exception! (filtering)" << std::endl;
413  std::cout << "i = " << i << std::endl;
414  throw std::runtime_error(e.what());
415  }
416 
417  return filtered;

608  }//end outer for
