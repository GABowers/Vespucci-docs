77 std::map<std::string, arma::mat> VespucciR::GetEnvironment(std::map<std::string, std::string> keys)
78 {
79  std::map<std::string, arma::mat> objects;
80  arma::mat matrix;
81  std::string vespucci_key;
82  std::string R_key;
83  for (std::map<std::string, std::string>::iterator i = keys.begin(); i != keys.end(); ++i){
84  vespucci_key = i->first;
85  R_key = i->second;
86  std::cout << "Get " << R_key << " as " << vespucci_key << std::endl;
87  try{
88  //breaking into parts because everything is really templated
89  R_key = "as.numeric(" + R_key + ")";
90  std::string query = "as.numeric(" + R_key + ")";
91  std::string existtest = "exists(\"" + R_key + "\")";
92  std::string mattest = "is.matrix(" + query + ")";
93  std::string vectest = "is.vector(" + query + ")";
94  bool exists = Rcpp::as<bool>(R_.parseEval(existtest));
95  bool ismat = Rcpp::as<bool>(R_.parseEval(mattest));
96  bool isvec = Rcpp::as<bool>(R_.parseEval(vectest));
97 
98  //vectors must be cast as arma::vec, matrices as arma::mat
99  //check if R object is matrix or vector
100  if (exists && ismat)
101  matrix = Rcpp::as<arma::mat>(R_.parseEval(query));
102  else if (exists && isvec)
103  matrix = (arma::mat) Rcpp::as<arma::vec>(R_.parseEval(query));
104  else
105  std::cerr << R_key << " is of unsupported type or does not exist!" << std::endl;
106 
107  }catch(std::exception e){
108  std::cerr << "Conversion to matrix failed" << std::endl;
109  }
110  try{
111  objects[vespucci_key] = matrix;
112  }catch(std::exception e){
113  std::cerr << "Adding " << R_key << " failed" << std::endl;
114  //ignore and move on.
115  }
116  }
117  return objects;
118 }

120 void VespucciR::RunScript(std::string cmd)
121 {
122  //read commands line by line for better error output
123  std::istringstream commands(cmd);
124  std::string command;
125  try{
126  while (!commands.eof()){
127  std::getline(commands, command);
128  std::cout << "R: " << command << std::endl;
129  try{
130  R_.parseEval(command);
131  }catch(std::exception e){
132  std::cerr << e.what();
133  std::string what(e.what());
134  what = "RInside Exception: " + what;
135  throw std::runtime_error(what.c_str());
136  }
137  }
138  }
139  catch (std::exception e)
140  {
141  std::cerr << e.what();
142  std::string what(e.what());
143  what = command;
144  throw std::runtime_error(what.c_str());
145  }
146 }
