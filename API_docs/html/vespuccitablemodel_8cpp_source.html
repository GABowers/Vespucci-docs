64 QVariant VespucciTableModel::data(const QModelIndex &index, int role) const
65 {
66  if (role != Qt::DisplayRole){
67  return QVariant();
68  }
69 
70  int row = index.row();
71  int column = index.column();
72  double value;
73  if (!composite_){
74  value = data_->at(row, column);
75  return QString::number(value);
76  }
77 
78  else
79  {
80  if (spatial_only_){
81  switch (column)
82  {
83  case 0:
84  return QString::number(dataset_->x_ptr()->at(row));
85  default:
86  return QString::number(dataset_->y_ptr()->at(row));
87  }
88 
89  }
90  switch (row)
91  {
92  case 0:
93  switch (column)
94  {
95  case 0:
96  case 1:
97  return "";
98  default:
99  return QString::number(dataset_->wavelength_ptr()->at(column));
100  }
101  default:
102  switch (column)
103  {
104  case 0:
105  return QString::number(dataset_->x_ptr()->at(row));
106  case 1:
107  return QString::number(dataset_->y_ptr()->at(row));
108  default:
109  return QString::number(dataset_->spectra_ptr()->at(row - 1, column - 2));
110  }
111 
112  }
113  }
