226 void MainWindow::on_actionNormalize_Standardize_triggered()
227 {
228  if (dataset_list_view_->model()->rowCount() < 1)
229  return;
230  int row = dataset_list_view_->currentIndex().row();
231  QSharedPointer<VespucciDataset> data = workspace->DatasetAt(row);
232  QStringList methods;
233  methods << "Min/Max" << "Vector Norm" << "Unit Area" << "Z-score"
234  << "Standard Normal Variate" << "Peak Intensity" << "Scale Spectra"
235  << "Absolute Value" << "Mean Center";
236  bool ok;
237  QString item = QInputDialog::getItem(this,
238  tr("Normalization/Standardization"),
239  tr("Method:"), methods, 0, false, &ok);
240  double scaling_factor, offset;
241  try{
242  if (ok && item == "Min/Max"){data->MinMaxNormalize();}
243  else if (ok && item == "Vector Norm"){data->VectorNormalize();}
244  else if (ok && item == "Mean Center"){data->MeanCenter();}
245  else if (ok && item == "Unit Area"){data->UnitAreaNormalize();}
246  else if (ok && item == "Z-score"){data->ZScoreNormalize();}
247  else if (ok && item == "Peak Intensity"){
248  double min = data->wavelength_ptr()->min();
249  double max = data->wavelength_ptr()->max();
250  RangeDialog *range_dialog = new RangeDialog(this, min, max);
251  QObject::connect(range_dialog, SIGNAL(DialogAccepted(double,double)), this, SLOT(RangeDialogAccepted(double,double)));
252  range_dialog->show();
253  }
254  else if (ok && item == "Scale Spectra"){
255  scaling_factor = QInputDialog::getDouble(this, tr("Enter Scaling Factor"),
256  tr("Factor"), 1, -100, 100, 2, &ok);
257  data->Scale(scaling_factor);
258  }
259  else if (ok && item == "Standard Normal Variate"){
260  offset = QInputDialog::getDouble(this, "Enter Offset", "Offset",
261  0, -2147483647, 2147483647, 4, &ok);
262  data->SNVNormalize(offset);
263  }
264  else if (ok && item == "Absolute Value"){data->AbsoluteValue();}
265  else
266  return;
267  }
268  catch(exception e){
269  DisplayExceptionWarning(e);
270  }
271 }

276 void MainWindow::on_actionSubtract_Background_triggered()
277 {
278  if (dataset_list_view_->model()->rowCount() < 1)
279  return;
280  int row = dataset_list_view_->currentIndex().row();
281  QSharedPointer<VespucciDataset> data = workspace->DatasetAt(row);
282  QString filename =
283  QFileDialog::getOpenFileName(this,
284  tr("Select Background File"),
285  workspace->directory(),
286  "Vespucci Spectrum Files (*.arma *.txt *.csv)");
287  mat input;
288  QFileInfo file_info(filename);
289  workspace->set_directory(file_info.dir().path());
290  bool success = input.load(filename.toStdString());
291  //We take spectra inputs as row-major
292  if (input.n_rows < input.n_cols){
293  try{
294  input = input.t();
295  }catch(exception e){
296  DisplayExceptionWarning(e);
297  }
298  }
299  //The first column is probably an abscissa.
300  if (input.n_cols > 2){
301  success = false;
302  }
303  if (input.n_cols == 2){
304  try{
305  input.shed_col(0);
306  }catch(exception e){
307  DisplayExceptionWarning(e);
308  }
309  }
310  if (!success){
311  QMessageBox::warning(this, "File Open Error", "File cannot be opened or contains improper data.");
312  return;
313  }
314  else{
315  try{
316  data->SubtractBackground(input, filename);
317  }
318  catch(exception e){
319  DisplayExceptionWarning(e);
320  }
321  }
322 }
